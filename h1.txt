# Basic Python Code Example
def greet(name):
  """This function greets the person passed in as a parameter."""
  return f"Hello, {name}!"

# Variables and Data Types
message = greet("state") # String assignment
number = 10             # Integer
is_python_fun = True    # Boolean

# Conditional Statement (if/else)
if number > 5:
  print(message)
else:
  print("Number is not greater than 5.")

# Loop (for loop with a list)
fruits = ["apple", "banana", "cherry"]
print("\nMy favorite fruits:")
for fruit in fruits:
  print(f"- {fruit.upper()}") # Using .upper() for uppercase

# Outputting results
print(f"\nFinal result: {message}")
